  $('a[href^=#]').click(function(){
    $('html, body').animate({
      scrollTop: $( $.attr(this, 'href') ).offset().top
    }, 1500, 'easeOutCubic');

    return false;

中文翻譯： 當點擊<a>元素中href屬性有#開頭的，執行功能，功能內容為<html>和<body>進行動畫效果，內容為 先抓要scroll到達物件的座標top值，然後用scrolltop設定要讓scrollbar到達的垂直高度，完成～

$('a[href^="#"]').on('click', function(event) {

    var target = $( $(this).attr('href') );

    if( target.length ) {
        event.preventDefault();
        $('html, body').animate({
            scrollTop: target.offset().top
        }, 1000);
    }

});
