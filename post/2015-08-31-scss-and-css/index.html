/*建立一個clearfix的mixin*/
@mixin clearfix {
  &:after, &:before {
    content: "";
    display: table;
  }
  &:after{
    clear:both;
  }
  *zoom:1;
}
 
/*在父元素include clearfix*/
.wrap{
  @include clearfix;
}
